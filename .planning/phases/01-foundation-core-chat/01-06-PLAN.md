Key patterns:
```python
import pytest
from unittest.mock import AsyncMock, patch, MagicMock
from remmeh.ui.app import RemmehApp
from remmeh.ui.widgets import MessageWidget
from remmeh.api.openrouter import RateLimitError, TimeoutError


# Mock OpenRouter stream to yield controlled tokens
async def mock_stream(*args, **kwargs):
    yield "Hello"
    yield " world"


@pytest.mark.asyncio
async def test_user_message_appears_in_chat(api_key, temp_db_path, monkeypatch):
    monkeypatch.setenv("REMMEH_DB_PATH", str(temp_db_path))
    with patch("remmeh.ui.screens.chat.OpenRouterClient") as MockClient:
        MockClient.return_value.stream_chat = mock_stream
        app = RemmehApp()
        async with app.run_test() as pilot:
            await pilot.click("#input-panel")
            await pilot.type("Hello there")
            await pilot.press("enter")
            await pilot.pause(0.5)  # let worker run
            # Check user message widget mounted
            messages = app.query(MessageWidget)
            user_msgs = [m for m in messages if m.role == "user"]
            assert len(user_msgs) == 1
            assert "Hello there" in user_msgs[0].full_content
```
