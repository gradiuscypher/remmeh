```python
def load_config() -> AppConfig:
    """Load config from environment. Raises ValueError if OPENROUTER_API_KEY is missing."""

        api_key=api_key,
        model=os.environ.get("REMMEH_DEFAULT_MODEL", DEFAULT_MODEL),
        db_path=os.environ.get("REMMEH_DB_PATH", DEFAULT_DB_PATH),
    )
```

```python
    """Base error for OpenRouter API failures."""
    def __init__(self, message: str, status_code: int | None = None): ...

class RateLimitError(OpenRouterError):
    """429 Too Many Requests from OpenRouter."""

class TimeoutError(OpenRouterError):
    """Request timed out."""

class NetworkError(OpenRouterError):
    """Connection or network failure."""

class OpenRouterClient:
    def __init__(self, api_key: str, base_url: str = OPENROUTER_BASE_URL): ...

    async def stream_chat(
        self,
        messages: list[dict],  # [{"role": "user", "content": "..."}]
        model: str,
        timeout: float = 60.0,
    ) -> AsyncIterator[str]:
        """Yield text tokens from a streaming chat completion.

        Raises:
            RateLimitError: on HTTP 429
            TimeoutError: on httpx.TimeoutException
            NetworkError: on httpx.ConnectError / httpx.RemoteProtocolError
            OpenRouterError: on other HTTP errors (4xx/5xx)
        """
```

**`remmeh/api/openrouter.py`** implementation notes:
- Use `httpx.AsyncClient` with `async with client.stream("POST", url, ...)` — this is the correct pattern for SSE streaming
- Parse SSE manually: iterate `response.aiter_lines()`, strip `data: ` prefix, skip `[DONE]`, parse JSON, extract `choices[0].delta.content`
- Wrap the entire stream in try/except to classify `httpx.TimeoutException`, `httpx.ConnectError`, `httpx.HTTPStatusError`
- For HTTP errors: check `response.status_code == 429` → `RateLimitError`, else → `OpenRouterError(msg, status_code=...)`
- Do NOT use `response.raise_for_status()` inside the stream context — handle errors manually to provide typed exceptions
- The `async def stream_chat(...)` must be an `AsyncIterator[str]` — use `yield` inside (makes it an async generator)
- Timeout: set `httpx.Timeout(timeout)` on the client; `connect=10.0` always

**Test structure:**

Use `mocker.AsyncMock` for `httpx.AsyncClient.__aenter__` stream context.
