Create `MessageWidget` — renders a single chat turn (user or assistant).

```python
from textual.widget import Widget
from textual.widgets import RichLog, Markdown, Static
from textual.app import ComposeResult
from remmeh.models import Message

class MessageWidget(Widget):
    """Renders a single chat message. Supports streaming via RichLog → Markdown swap."""

    DEFAULT_CSS = """
    MessageWidget {
        margin: 0 0 1 0;
        padding: 0 1;
    }
    MessageWidget.user {
        border-left: solid $primary;
        padding-left: 2;
    }
    MessageWidget.assistant {
        padding-left: 0;
    }
    MessageWidget.error {
        border-left: solid $error;
        padding-left: 2;
    }
    .message-header {
        text-style: bold;
        margin-bottom: 1;
    }
    .user .message-header {
        color: $primary;
    }
    .error .message-header {
        color: $error;
    }
    """

    def __init__(self, role: str, initial_content: str = "", **kwargs) -> None:
        super().__init__(classes=role, **kwargs)
        self.role = role
        self._content = initial_content
        self._streaming = role == "assistant" and not initial_content
        self._rich_log: RichLog | None = None

    def compose(self) -> ComposeResult:
        # Header: "You:", "Assistant:", or "[bold $error]Error:[/]"
        if self.role == "user":
            yield Static("[bold]You:[/]", classes="message-header")
            yield Static(self._content)
        elif self.role == "assistant":
            yield Static("[bold]Assistant:[/]", classes="message-header")
            if self._streaming:
                # Streaming mode: RichLog for incremental writes (pitfall #2)
                # Do NOT use Markdown widget during streaming — it re-renders everything
                rich_log = RichLog(id="stream-log", wrap=True, markup=False)
                self._rich_log = rich_log
                yield rich_log
            else:
                # Completed response: use Markdown for full rendering
                yield Markdown(self._content)
        elif self.role == "error":
            yield Static("[bold $error]Error:[/]", classes="message-header")
            yield Static(self._content)

    def write_chunk(self, chunk: str) -> None:
        """Called per streaming token. Writes to RichLog (no scroll reset)."""
        if self._rich_log is not None:
            self._rich_log.write(chunk, end="")
        self._content += chunk

    def swap_to_markdown(self) -> None:
        """Replace RichLog with Markdown widget on stream completion.

        Streaming swap pattern (pitfall #2):
        - RichLog.write(chunk) per token during streaming
        - Replace with Markdown(full_text) on completion
        - Do NOT call Markdown.update() per chunk
        """
        if self._rich_log is not None:
            # Remove RichLog, mount Markdown with full accumulated content
            rich_log = self._rich_log
            self._rich_log = None
            self._streaming = False
            # Use call_after_refresh to avoid mid-render mutation
            self.app.call_after_refresh(self._do_swap, rich_log)

    def _do_swap(self, rich_log: RichLog) -> None:
        rich_log.remove()
        self.mount(Markdown(self._content))

    @property
    def full_content(self) -> str:
        return self._content
```

```python
from __future__ import annotations

import asyncio
from textual.screen import Screen
from textual.app import ComposeResult
from textual.worker import Worker, WorkerState
from textual import work

from remmeh.ui.widgets import ChatView, InputPanel, StatusBar, MessageWidget
from remmeh.api.openrouter import OpenRouterClient, OpenRouterError, RateLimitError, TimeoutError, NetworkError
from remmeh.config import load_config, DEFAULT_MODEL
from remmeh.models import Message, ChatSession
from remmeh.storage import append_message, create_session, init_db


class ChatScreen(Screen):
    """Primary screen: StatusBar / ChatView / InputPanel."""

    CSS_PATH = "../remmeh.tcss"

    BINDINGS = [
        ("ctrl+q", "quit", "Quit"),
        ("ctrl+c", "quit", "Quit"),
        ("escape", "cancel_stream", "Cancel stream"),
    ]

    def __init__(self) -> None:
        super().__init__()
        # Config loaded lazily on first submit to give a good error message
        self._config = None
        self._client: OpenRouterClient | None = None
        self._session: ChatSession | None = None
        self._current_assistant_widget: MessageWidget | None = None
        self._stream_messages: list[dict] = []  # OpenRouter message history

    def compose(self) -> ComposeResult:
        yield StatusBar(id="status-bar")
        yield ChatView(id="chat-view")
        yield InputPanel(id="input-panel")

    def on_mount(self) -> None:
        """Initialize status bar with default model."""
        self.query_one(StatusBar).update_model(DEFAULT_MODEL)
        # DB init happens here to ensure it's ready before first submit
        # Uses asyncio task to avoid blocking the UI mount
        self.run_worker(self._init_storage(), exclusive=False)

    async def _init_storage(self) -> None:
        """Initialize SQLite DB with WAL mode at startup."""
        try:
            cfg = self._get_config()
            import os
            from pathlib import Path
            Path(cfg.db_path).parent.mkdir(parents=True, exist_ok=True)
            await init_db(cfg.db_path)
        except ValueError:
            pass  # API key not set yet — will error on first submit, not on startup

    def _get_config(self):
        """Lazy config load — raises ValueError with good message if key missing."""
        if self._config is None:
            self._config = load_config()
            self._client = OpenRouterClient(api_key=self._config.api_key)
            self.query_one(StatusBar).update_model(self._config.model)
        return self._config

    async def _ensure_session(self) -> ChatSession:
        """Create a new ChatSession in SQLite if none exists."""
        if self._session is None:
            cfg = self._get_config()
            session = ChatSession(model=cfg.model)
            self._session = await create_session(cfg.db_path, session)
        return self._session

    def on_input_panel_submitted(self, event: InputPanel.Submitted) -> None:
        """Handle user message submission."""
        text = event.text.strip()
        if not text:
            return
        self._submit_message(text)

    @work(exclusive=True, exit_on_error=False)
    async def _submit_message(self, text: str) -> None:
        """Streaming chat worker.

        CRITICAL PITFALLS ADDRESSED:
        1. @work(exclusive=True) — submitting a new message cancels any in-progress stream
        3. exit_on_error=False — errors don't kill the TUI; handled in on_worker_state_changed
        """
        chat_view = self.query_one(ChatView)
        input_panel = self.query_one(InputPanel)
        status_bar = self.query_one(StatusBar)

        # --- Pre-flight: validate config ---
        try:
            cfg = self._get_config()
        except ValueError as exc:
            # API key missing — show inline error
            error_widget = MessageWidget(role="error", initial_content=str(exc))
            chat_view.add_message_widget(error_widget)
            return

        # --- Show user message in UI ---
        user_widget = MessageWidget(role="user", initial_content=text)
        chat_view.add_message_widget(user_widget)

        # --- Build message history for API ---
        self._stream_messages.append({"role": "user", "content": text})

        # --- Save user message to SQLite ---
        session = await self._ensure_session()
        user_msg = Message(role="user", content=text, session_id=session.id)
        await append_message(cfg.db_path, user_msg)

        # --- Disable input and show streaming state ---
        input_panel.disable()
        status_bar.set_streaming(True)

        # --- Create assistant widget in streaming mode ---
        assistant_widget = MessageWidget(role="assistant")
        self._current_assistant_widget = assistant_widget
        chat_view.add_message_widget(assistant_widget)

        # --- Stream tokens ---
        full_response = ""
        try:
            async for chunk in self._client.stream_chat(
                messages=self._stream_messages,
                model=cfg.model,
            ):
                assistant_widget.write_chunk(chunk)
                full_response += chunk
                chat_view.scroll_to_end()
        except RateLimitError:
            error_msg = "Rate limit reached. Wait a moment and try again."
            self._replace_with_error(assistant_widget, chat_view, error_msg)
            full_response = ""
        except TimeoutError:
            error_msg = "Request timed out. Check your connection and try again."
            self._replace_with_error(assistant_widget, chat_view, error_msg)
            full_response = ""
        except NetworkError:
            error_msg = "Connection error. Check your internet connection and try again."
            self._replace_with_error(assistant_widget, chat_view, error_msg)
            full_response = ""
        except OpenRouterError as exc:
            error_msg = f"Request failed: {exc}. Try again or check your API key."
            self._replace_with_error(assistant_widget, chat_view, error_msg)
            full_response = ""
        except asyncio.CancelledError:
            # Stream cancelled by Escape key — leave partial response as-is
            # No error widget — cancellation is intentional
            pass
        else:
            # Stream completed successfully — swap RichLog → Markdown
            assistant_widget.swap_to_markdown()
            # Update message history for context
            if full_response:
                self._stream_messages.append({"role": "assistant", "content": full_response})

        # --- Save assistant response to SQLite (if any content) ---
        if full_response:
            asst_msg = Message(role="assistant", content=full_response, session_id=session.id)
            await append_message(cfg.db_path, asst_msg)

        # --- Re-enable input and clear streaming state ---
        input_panel.enable()
        status_bar.set_streaming(False)
        self._current_assistant_widget = None

    def _replace_with_error(
        self,
        assistant_widget: MessageWidget,
        chat_view: ChatView,
        error_msg: str,
    ) -> None:
        """Replace a streaming assistant widget with an error widget."""
        assistant_widget.remove()
        error_widget = MessageWidget(role="error", initial_content=error_msg)
        chat_view.add_message_widget(error_widget)

    def action_cancel_stream(self) -> None:
        """Cancel active streaming worker (Escape key)."""
        # workers attribute from Textual — cancel all workers on this screen
        for worker in self.workers:
            if not worker.is_done:
                worker.cancel()
```

**Implementation notes:**
- `@work(exclusive=True, exit_on_error=False)` is the ONLY safe pattern for async LLM calls in Textual (pitfall #1 + #3)
- `exit_on_error=False` prevents unhandled exceptions from killing the TUI
- `asyncio.CancelledError` is re-raised by `@work` when cancelled — catch it to handle graceful cancel
- `_replace_with_error` mounts error widget in place of the streaming assistant widget
- Do NOT use `self.call_later()` or `asyncio.create_task()` directly — use `@work`
